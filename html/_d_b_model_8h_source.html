228 void del(int pos)
229 {
230 // Start position of chunk
231 long int sbytes = pos * sizeof(T); //Start byte of record to del.
232 long int ebytes = size(); //Size (bytes) of file
233 long int rbytes = sizeof(T); //Size (bytes) of record
234 long int cbytes = ebytes - sbytes - rbytes; //Size (bytes) of chunk to shift
235
236 char buffer_a[sbytes];
237 char buffer_b[cbytes];
238 // Read the chunk of bytes up to record to be deleted into buffer
239 file.seekg(0L, ios::beg);
240 file.read(buffer_a, sbytes);
241 // Read the chunk of bytes after record to be deleted into the buffer
242 file.seekg(sbytes + rbytes, ios::beg);
243 file.read(buffer_b, cbytes);
244
245 // Clear file contents;
246 delAll();
247
248 // Reconstruct file contents (without deleted record)
249 file.seekp(0L, ios::end);
250 file.write(buffer_a, sbytes);
251 file.write(buffer_b, cbytes);
252
253 file.flush();
254 }
